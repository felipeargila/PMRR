        function selecionarFiltro(tipo, checkbox) {

            if (checkbox.checked) {

                filtrosAtivos[tipo].push(checkbox.value);

            } else {

                const index = filtrosAtivos[tipo].indexOf(checkbox.value);

                if (index > -1) filtrosAtivos[tipo].splice(index, 1);

            }

            atualizarChips();

        }


        function atualizarChips() {

            const container = document.getElementById('container-chips');

            let html = '<div class="flex flex-wrap items-center gap-2">';

            

            Object.keys(filtrosAtivos).forEach(tipo => {

                filtrosAtivos[tipo].forEach(valor => {

                    html += `<span class="chip">

                        ${valor} 

                        <span class="chip-remover" onclick="removerFiltro('${tipo}', '${valor}')">

                            <i class="fas fa-times"></i>

                        </span>

                    </span>`;

                });

            });

            

            if (filtrosAtivos.estados.length + filtrosAtivos.riscos.length + filtrosAtivos.status.length + filtrosAtivos.tipos.length > 0) {

                html += `<span class="text-xs text-blue-600 ml-2 cursor-pointer hover:underline" onclick="limparTodosFiltros()">Limpar todos</span>`;

            } else {

                html += '<span class="text-xs text-gray-400">Nenhum filtro aplicado</span>';

            }

            

            html += '</div>';

            container.innerHTML = html;

        }


        function removerFiltro(tipo, valor) {

            const index = filtrosAtivos[tipo].indexOf(valor);

            if (index > -1) {

                filtrosAtivos[tipo].splice(index, 1);

                document.querySelectorAll('.filtro-checkbox').forEach(cb => {

                    if (cb.value === valor) cb.checked = false;

                });

                atualizarChips();

                aplicarFiltros();

            }

        }


        function limparTodosFiltros() {

            filtrosAtivos = { estados: [], riscos: [], status: [], tipos: [] };

            document.querySelectorAll('.filtro-checkbox').forEach(cb => cb.checked = false);

            atualizarChips();

            aplicarFiltros();

        }


        function toggleFiltro(tipo) {

            const content = document.getElementById(`filtro-${tipo}-content`);

            content.classList.toggle('hidden');

        }


        function toggleFiltros() {

            const filtros = document.getElementById('filtrosSidebar');

            const textoBotao = document.getElementById('textoBotaoFiltros');

            

            if (filtros.classList.contains('hidden')) {

                filtros.classList.remove('hidden');

                filtros.classList.add('block');

                textoBotao.textContent = 'Ocultar filtros';

            } else {

                filtros.classList.add('hidden');

                filtros.classList.remove('block');

                textoBotao.textContent = 'Mostrar filtros';

            }

        }


        function aplicarFiltros() {

            municipiosFiltrados = municipios.filter(m => {

                if (filtrosAtivos.estados.length > 0 && !filtrosAtivos.estados.includes(m.uf)) return false;

                if (filtrosAtivos.riscos.length > 0 && !filtrosAtivos.riscos.includes(m.risco)) return false;

                if (filtrosAtivos.status.length > 0 && !filtrosAtivos.status.includes(m.status)) return false;

                if (filtrosAtivos.tipos.length > 0 && !filtrosAtivos.tipos.includes(m.tipoRisco)) return false;

                return true;

            });

            

            pagina = 1;

            atualizarInterface();

            if (window.innerWidth < 1024) {

                document.getElementById('filtrosSidebar').classList.add('hidden');

                document.getElementById('textoBotaoFiltros').textContent = 'Mostrar filtros';

            }

        }

        function atualizarMapa() {

            if (!map) return;

            

            map.eachLayer((layer) => {

                if (layer instanceof L.Marker || layer instanceof L.CircleMarker) {

                    map.removeLayer(layer);

                }

            });


            municipiosFiltrados.slice(0, 100).forEach(m => {

                if (m.lat && m.lng) {

                    const cor = m.risco === 'Alto' ? '#EF4444' : m.risco === 'Médio' ? '#F59E0B' : '#10B981';

                    L.circleMarker([m.lat, m.lng], {

                        radius: 5,

                        fillColor: cor,

                        color: '#fff',

                        weight: 2,

                        opacity: 1,

                        fillOpacity: 0.8

                    })

                    .addTo(map)

                    .bindPopup(`

                        <b>${m.nome} - ${m.uf}</b><br>

                        Risco: ${m.risco}<br>

                        Status: ${m.status}<br>

                        ${m.tipoRisco}

                    `);

                }

            });

        }


        function ordenarMunicipios() {

            const criterio = document.getElementById('ordenarPor').value;

            

            switch(criterio) {

                case 'nome':

                    municipiosFiltrados.sort((a, b) => a.nome.localeCompare(b.nome));

                    break;

                case 'risco':

                    const ordemRisco = { 'Alto': 1, 'Médio': 2, 'Baixo': 3 };

                    municipiosFiltrados.sort((a, b) => ordemRisco[a.risco] - ordemRisco[b.risco]);

                    break;

                case 'data':

                    municipiosFiltrados.sort((a, b) => b.data.localeCompare(a.data));

                    break;

            }

            

            renderizarCards();

        }


        function mudarPagina(direcao) {

            const totalPaginas = Math.ceil(municipiosFiltrados.length / itemsPorPagina);

            if (direcao === 'proxima' && pagina < totalPaginas) {

                pagina++;

                renderizarCards();

            } else if (direcao === 'anterior' && pagina > 1) {

                pagina--;

                renderizarCards();

            }

        }
